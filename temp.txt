import React, { useState, useEffect, useRef } from "react";
import { motion, AnimatePresence } from "framer-motion";
import {
  Camera, Check, ChevronDown, Trash2, ArrowUp, ArrowDown,
  User, Info, Github, Linkedin, Twitter, Instagram, 
  Facebook, Youtube, Globe, MapPin, Lock, Unlock, Edit2, LinkIcon, Plus, Loader2
} from "lucide-react";
import { toast } from "sonner";
import {
  Dialog, DialogContent, DialogTrigger,
} from "./dialog";
import { Input } from "./input";
import { Label } from "./label";
import { Textarea } from "./textarea";
import { Button } from "./button";

type School = { id: number; name: string; short_name: string; type: string; region: string; logo: string; };

const SOCIAL_PLATFORMS = [
  { id: "github", label: "GitHub", icon: Github, color: "text-slate-800" },
  { id: "linkedin", label: "LinkedIn", icon: Linkedin, color: "text-blue-600" },
  { id: "twitter", label: "Twitter", icon: Twitter, color: "text-sky-500" },
  { id: "instagram", label: "Instagram", icon: Instagram, color: "text-pink-600" },
  { id: "facebook", label: "Facebook", icon: Facebook, color: "text-blue-700" },
  { id: "tiktok", label: "TikTok", icon: Globe, color: "text-black" },
  { id: "youtube", label: "YouTube", icon: Youtube, color: "text-red-600" },
  { id: "discord", label: "Discord", icon: Globe, color: "text-indigo-500" },
  { id: "website", label: "Website", icon: LinkIcon, color: "text-slate-600" },
];

export function ProfileSection() {
  const [isPrivate, setIsPrivate] = useState(false);
  const [isDialogOpen, setIsDialogOpen] = useState(false);

  const [profile, setProfile] = useState({
    displayName: "Alex Student",
    pronouns: "they/them",
    school: { id: 0, name: "University of Technology", short_name: "UoT", logo: "", type: "University" },
    bio: "Computer Science major passionate about AI, design systems, and making education more accessible.",
    avatarUrl: "https://api.dicebear.com/7.x/avataaars/svg?seed=Alex",
    socials: [
      { id: "1", platform: "github", url: "https://github.com/alex" },
      { id: "2", platform: "linkedin", url: "https://linkedin.com/in/alex" }
    ]
  });

  const [editForm, setEditForm] = useState(profile);

  // Avatar Upload State
  const [isUploading, setIsUploading] = useState(false);
  const [uploadProgress, setUploadProgress] = useState(0);
  const fileInputRef = useRef<HTMLInputElement>(null);

  // Schools ComboBox State
  const [schools, setSchools] = useState<School[]>([]);
  const [schoolSearch, setSchoolSearch] = useState("");
  const [isSchoolBoxOpen, setIsSchoolBoxOpen] = useState(false);
  const searchRef = useRef<HTMLDivElement>(null);

  // Socials
  const [isSocialSelectOpen, setIsSocialSelectOpen] = useState(false);

  useEffect(() => {
    import("./ph_schools.json").then((mod) => {
      setSchools(mod.default || mod);
    }).catch(err => console.error("Failed to load schools:", err));
  }, []);

  useEffect(() => {
    if (isDialogOpen) {
      setEditForm(profile);
      setSchoolSearch(profile.school?.name || "");
    }
  }, [isDialogOpen, profile]);

  useEffect(() => {
    const handleClickOutside = (e) => {
      if (searchRef.current && !searchRef.current.contains(e.target)) {
        setIsSchoolBoxOpen(false);
      }
    };
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);
  
  const handleAvatarUpload = (e) => {
    if (e.target.files && e.target.files[0]) {
      setIsUploading(true);
      setUploadProgress(0);
      let prog = 0;
      const int = setInterval(() => {
        prog += 20;
        setUploadProgress(prog);
        if(prog >= 100) {
          clearInterval(int);
          setIsUploading(false);
          setEditForm(prev => ({...prev, avatarUrl: "https://api.dicebear.com/7.x/avataaars/svg?seed=" + Math.random().toString(36).substring(7)}));
          toast.success("Profile picture updated! ");
        }
      }, 300);
    }
  };

  const handleSave = () => {
    setProfile(editForm);
    setIsDialogOpen(false);
    toast.success("Profile updated successfully!");
  };

  const filteredSchools = schools.filter(s => 
    s.name.toLowerCase().includes(schoolSearch.toLowerCase()) || 
    s.short_name.toLowerCase().includes(schoolSearch.toLowerCase())
  ).slice(0, 10);

  const addSocial = (platformId) => {
    const newId = Math.random().toString(36).substr(2, 9);
    setEditForm(prev => ({
      ...prev,
      socials: [...prev.socials, { id: newId, platform: platformId, url: "" }]
    }));
    setIsSocialSelectOpen(false);
  };

  const updateSocialUrl = (id, url) => {
    setEditForm(prev => ({
      ...prev,
      socials: prev.socials.map(s => s.id === id ? { ...s, url } : s)
    }));
  };

  const removeSocial = (id) => {
    setEditForm(prev => ({
      ...prev,
      socials: prev.socials.filter(s => s.id !== id)
    }));
  };

  const moveSocial = (index, direction) => {
    const newSocials = [...editForm.socials];
    if (direction === "up" && index > 0) {
      [newSocials[index - 1], newSocials[index]] = [newSocials[index], newSocials[index - 1]];
    } else if (direction === "down" && index < newSocials.length - 1) {
      [newSocials[index + 1], newSocials[index]] = [newSocials[index], newSocials[index + 1]];
    }
    setEditForm(prev => ({...prev, socials: newSocials}));
  };

  return (
    <div className="flex flex-col bg-white rounded-[1.5rem] border border-slate-200/60 shadow-sm p-5 sm:p-6 w-full h-full overflow-y-auto scrollbar-thin scrollbar-thumb-slate-200">
      <div className="flex items-center justify-between mb-5">
        <h2 className="text-lg font-bold text-slate-800">My Profile</h2>
        <div className="flex items-center gap-2.5">
          <span className="text-xs font-bold text-slate-500 uppercase tracking-widest">
            {isPrivate ? "Private" : "Public"}
          </span>
          <button
            onClick={() => setIsPrivate(!isPrivate)}
            className={`relative inline-flex h-6 w-11 items-center rounded-full transition-colors focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2 ${isPrivate ? "bg-indigo-600" : "bg-slate-200"}`}
          >
            <span className={`inline-block h-4 w-4 transform rounded-full bg-white transition-transform ${isPrivate ? "translate-x-6" : "translate-x-1"}`} />
          </button>
        </div>
      </div>

      <div className="flex flex-col mb-8 gap-6">
        <div className="flex flex-row items-start gap-5">
          <div className="relative shrink-0">
            <div className="w-[120px] h-[120px] rounded-full overflow-hidden border-4 border-white shadow-lg bg-orange-100">
              <img src={profile.avatarUrl} alt="Avatar" className="w-full h-full object-cover" />
            </div>
            {isPrivate && (
              <div className="absolute bottom-1 right-1 bg-slate-800 text-white p-1.5 rounded-full shadow-sm border-2 border-white">
                <Lock className="w-4 h-4" />
              </div>
            )}
          </div>
          <div className="flex flex-col pt-2">
            <div className="flex flex-row items-center gap-3 mb-1">
              <h3 className="text-2xl font-bold text-slate-800">{profile.displayName}</h3>
              <span className="px-2.5 py-1 bg-slate-100 text-slate-600 text-xs font-semibold rounded-full">
                {profile.pronouns}
              </span>
            </div>
            <div className="flex items-center gap-1.5 text-sm text-indigo-600 font-semibold mb-3">
              {profile.school.logo ? <img src={profile.school.logo} alt="logo" className="w-4 h-4 object-contain" /> : <MapPin className="w-4 h-4" />}
              <span>{profile.school.name}</span>
            </div>
            <p className="text-sm text-slate-600 leading-relaxed max-w-md">
              {profile.bio}
            </p>
          </div>
        </div>

        {profile.socials.length > 0 && (
          <div className="border-t border-slate-100 pt-5">
            <h4 className="text-xs font-bold text-slate-500 uppercase tracking-widest mb-3">Digital Footprint</h4>
            <div className="flex flex-row flex-wrap gap-2">
              {profile.socials.map(social => {
                const platformInfo = SOCIAL_PLATFORMS.find(p => p.id === social.platform);
                if (!platformInfo) return null;
                const Icon = platformInfo.icon;
                return (
                  <a
                    key={social.id}
                    href={social.url}
                    target="_blank"
                    rel="noopener noreferrer"
                    className="flex items-center gap-2 px-3 py-1.5 bg-slate-50 hover:bg-slate-100 border border-slate-200 rounded-lg transition-colors group"
                  >
                    <Icon className={`w-4 h-4 ${platformInfo.color} group-hover:scale-110 transition-transform`} />
                    <span className="text-sm font-medium text-slate-700">{platformInfo.label}</span>
                  </a>
                );
              })}
            </div>
          </div>
        )}

        <div className="border-t border-slate-100 pt-5 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="w-10 h-10 rounded-full bg-emerald-50 border border-emerald-100 flex items-center justify-center">
              <Check className="w-5 h-5 text-emerald-600" />
            </div>
            <div>
              <div className="text-sm font-bold text-slate-800">Profile Completion</div>
              <div className="text-xs font-medium text-emerald-600">85% <span className="text-slate-400 font-normal">&bull; Last updated today</span></div>
            </div>
          </div>
        </div>
      </div>

      <Dialog open={isDialogOpen} onOpenChange={setIsDialogOpen}>
        <DialogTrigger asChild>
          <Button className="w-full bg-indigo-50 hover:bg-indigo-100 text-indigo-600 font-semibold rounded-xl h-11 shadow-sm border border-indigo-100/50 transition-all">
            <Edit2 className="w-4 h-4 mr-2" />
            Edit Profile
          </Button>
        </DialogTrigger>
        
        <DialogContent className="max-w-[900px] h-[750px] p-0 gap-0 overflow-y-auto rounded-[2.5rem] bg-white flex flex-col shadow-[0_20px_50px_rgba(0,0,0,0.1)] border-0">
          <div className="relative h-32 bg-gradient-to-r from-indigo-500 via-purple-500 to-pink-500 shrink-0">
            <div className="absolute inset-0 bg-white/10 backdrop-blur-[1px]"></div>
          </div>
          
          <div className="px-8 sm:px-12 pb-24 pt-0 w-full max-w-3xl mx-auto flex-1 flex flex-col">
            
            {/* Interactive Profile Picture */}
            <div className="relative -mt-16 mb-10 flex justify-center">
              <input type="file" ref={fileInputRef} className="hidden" accept="image/*" onChange={handleAvatarUpload} />
              
              <div 
                className="group relative w-[150px] h-[150px] rounded-full overflow-hidden shadow-xl border-[6px] border-white bg-slate-100 cursor-pointer flex-shrink-0 transition-transform hover:scale-105 active:scale-95"
                onClick={() => !isUploading && fileInputRef.current?.click()}
              >
                <img src={editForm.avatarUrl} alt="Edit Avatar" className="w-full h-full object-cover" />
                
                <div className="absolute inset-0 bg-black/40 flex items-center justify-center opacity-0 group-hover:opacity-100 transition-opacity">
                  <Camera className="w-8 h-8 text-white" />
                </div>

                {isUploading && (
                  <div className="absolute inset-0 bg-white/80 flex flex-col items-center justify-center">
                    <Loader2 className="w-8 h-8 text-indigo-600 animate-spin mb-2" />
                    <span className="text-xs font-bold text-indigo-700">{uploadProgress}%</span>
                    <motion.div 
                      className="absolute bottom-0 left-0 h-1.5 bg-indigo-600" 
                      initial={{ width: 0 }}
                      animate={{ width: `${uploadProgress}%` }}
                      transition={{ duration: 0.2 }}
                    />
                  </div>
                )}
              </div>
            </div>

            <div className="space-y-12">
              
              {/* Section 1: Personal Info */}
              <section className="space-y-6">
                <div className="border-b border-slate-200 pb-2">
                  <h2 className="text-xl font-bold tracking-tight text-slate-800">Personal Information</h2>
                  <p className="text-sm text-slate-500">Your basic details and educational background.</p>
                </div>
                
                <div className="grid grid-cols-1 sm:grid-cols-2 gap-6">
                  <div className="space-y-2">
                    <Label className="text-slate-700 font-semibold">Full Name</Label>
                    <Input 
                      value={editForm.displayName} 
                      onChange={e => setEditForm({...editForm, displayName: e.target.value})} 
                      className="h-12 rounded-xl bg-slate-50 border-slate-200 focus:bg-white focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100 transition-all"
                      placeholder="e.g. Jane Doe"
                    />
                  </div>
                  <div className="space-y-2">
                    <Label className="text-slate-700 font-semibold">Pronouns</Label>
                    <Input 
                      value={editForm.pronouns} 
                      onChange={e => setEditForm({...editForm, pronouns: e.target.value})} 
                      className="h-12 rounded-xl bg-slate-50 border-slate-200 focus:bg-white focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100 transition-all"
                      placeholder="e.g. she/her"
                    />
                  </div>
                </div>

                <div className="space-y-2 relative" ref={searchRef}>
                  <Label className="text-slate-700 font-semibold">School / University</Label>
                  <div className="relative">
                    <MapPin className="absolute left-4 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
                    <Input 
                      value={schoolSearch}
                      onChange={e => {
                        setSchoolSearch(e.target.value);
                        setIsSchoolBoxOpen(true);
                      }}
                      onFocus={() => setIsSchoolBoxOpen(true)}
                      className="h-12 pl-10 rounded-xl bg-slate-50 border-slate-200 focus:bg-white focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100 transition-all"
                      placeholder="Search for your school..."
                    />
                  </div>
                  
                  {isSchoolBoxOpen && schoolSearch && (
                    <motion.div 
                      key="school-dropdown"
                      initial={{ opacity: 0, y: -10 }}
                      animate={{ opacity: 1, y: 0 }}
                      className="absolute z-50 w-full mt-2 bg-white rounded-xl shadow-xl border border-slate-100 max-h-60 overflow-y-auto"
                    >
                      {filteredSchools.length > 0 ? (
                        <div className="p-2 space-y-1">
                          {filteredSchools.map(school => (
                            <button
                              key={school.id}
                              onClick={() => {
                                setEditForm({...editForm, school});
                                setSchoolSearch(school.name);
                                setIsSchoolBoxOpen(false);
                              }}
                              className="w-full flex items-center gap-3 p-3 hover:bg-slate-50 rounded-lg text-left transition-colors"
                            >
                              {school.logo ? (
                                <img src={school.logo} alt="" className="w-10 h-10 object-contain rounded bg-white p-1 border border-slate-100" />
                              ) : (
                                <div className="w-10 h-10 rounded bg-slate-100 flex items-center justify-center border border-slate-200">
                                  <MapPin className="w-4 h-4 text-slate-400" />
                                </div>
                              )}
                              <div className="flex-1 min-w-0">
                                <p className="text-sm font-bold text-slate-800 truncate">{school.name}</p>
                                <div className="flex gap-2 items-center text-xs text-slate-500">
                                  <span className="font-medium text-indigo-600">{school.short_name}</span>
                                  <span>&bull;</span>
                                  <span>{school.type}</span>
                                </div>
                              </div>
                            </button>
                          ))}
                        </div>
                      ) : (
                        <div className="p-4 text-center text-sm text-slate-500">
                          No schools found. Try a different search term.
                        </div>
                      )}
                    </motion.div>
                  )}
                </div>
              </section>

              {/* Section 2: About */}
              <section className="space-y-6">
                <div className="border-b border-slate-200 pb-2">
                  <h2 className="text-xl font-bold tracking-tight text-slate-800">About Me</h2>
                  <p className="text-sm text-slate-500">A brief description of who you are.</p>
                </div>
                
                <div className="space-y-2">
                  <Textarea 
                    value={editForm.bio}
                    onChange={e => {
                      if(e.target.value.length <= 280) {
                        setEditForm({...editForm, bio: e.target.value});
                      }
                    }}
                    className="min-h-[120px] rounded-xl bg-slate-50 border-slate-200 focus:bg-white focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100 transition-all resize-none p-4"
                    placeholder="Write something about yourself..."
                  />
                  <div className="flex justify-end">
                    <span className={`text-xs font-semibold ${editForm.bio.length >= 280 ? 'text-red-500' : 'text-slate-400'}`}>
                      {editForm.bio.length} / 280
                    </span>
                  </div>
                </div>
              </section>

              {/* Section 3: Digital Footprint */}
              <section className="space-y-6 flex flex-col mb-4">
                <div className="border-b border-slate-200 pb-2 flex justify-between items-end">
                  <div>
                    <h2 className="text-xl font-bold tracking-tight text-slate-800">Digital Footprint</h2>
                    <p className="text-sm text-slate-500">Your presence on the web.</p>
                  </div>
                  
                  <div className="relative">
                    <Button 
                      variant="outline" 
                      size="sm" 
                      onClick={() => setIsSocialSelectOpen(!isSocialSelectOpen)}
                      className="rounded-full font-semibold border-slate-200 hover:bg-slate-50 text-indigo-600"
                    >
                      <Plus className="w-4 h-4 mr-1.5" /> Add Social
                    </Button>
                    
                    {isSocialSelectOpen && (
                      <div className="absolute right-0 top-full mt-2 w-56 bg-white border border-slate-200 rounded-xl shadow-lg z-10 py-2">
                        <div className="max-h-60 overflow-y-auto">
                          {SOCIAL_PLATFORMS.map(platform => {
                            const Icon = platform.icon;
                            const isAdded = editForm.socials.some(s => s.platform === platform.id);
                            if (isAdded) return null;
                            return (
                              <button
                                key={platform.id}
                                onClick={() => addSocial(platform.id)}
                                className="w-full flex items-center gap-3 px-4 py-2 hover:bg-slate-50 text-left text-sm font-medium text-slate-700 transition-colors"
                              >
                                <Icon className={`w-4 h-4 ${platform.color}`} />
                                {platform.label}
                              </button>
                            );
                          })}
                        </div>
                      </div>
                    )}
                  </div>
                </div>

                <div className="space-y-3">
                  <AnimatePresence>
                    {editForm.socials.map((social, index) => {
                      const platformInfo = SOCIAL_PLATFORMS.find(p => p.id === social.platform);
                      if (!platformInfo) return null;
                      const Icon = platformInfo.icon;
                      
                      return (
                        <motion.div
                          key={social.id}
                          initial={{ opacity: 0, scale: 0.95 }}
                          animate={{ opacity: 1, scale: 1 }}
                          exit={{ opacity: 0, height: 0, marginBottom: 0 }}
                          className="flex flex-col sm:flex-row items-stretch sm:items-center gap-3 p-4 bg-slate-50 rounded-xl border border-slate-100 group"
                        >
                          <div className="flex items-center gap-3 w-full sm:w-auto">
                            <div className="flex flex-col gap-0.5 mr-2 sm:mr-0">
                              <button onClick={() => moveSocial(index, "up")} disabled={index === 0} className="text-slate-400 hover:text-indigo-600 disabled:opacity-30">
                                <ArrowUp className="w-4 h-4" />
                              </button>
                              <button onClick={() => moveSocial(index, "down")} disabled={index === editForm.socials.length - 1} className="text-slate-400 hover:text-indigo-600 disabled:opacity-30">
                                <ArrowDown className="w-4 h-4" />
                              </button>
                            </div>
                            
                            <div className="w-10 h-10 rounded-full bg-white flex items-center justify-center shadow-sm shrink-0">
                              <Icon className={`w-5 h-5 ${platformInfo.color}`} />
                            </div>
                            <span className="font-bold text-slate-700 w-24 hidden sm:block">{platformInfo.label}</span>
                          </div>
                          
                          <div className="flex-1 flex gap-3">
                            <Input 
                              value={social.url}
                              onChange={e => updateSocialUrl(social.id, e.target.value)}
                              placeholder={`https://${platformInfo.id}.com/username`}
                              className="h-10 rounded-lg bg-white border-slate-200 focus:border-indigo-400 focus:ring-4 focus:ring-indigo-100 flex-1"
                            />
                            <Button 
                              variant="ghost" 
                              size="icon" 
                              onClick={() => removeSocial(social.id)}
                              className="h-10 w-10 text-slate-400 hover:text-red-500 hover:bg-red-50 shrink-0"
                            >
                              <Trash2 className="w-4 h-4" />
                            </Button>
                          </div>
                        </motion.div>
                      );
                    })}
                  </AnimatePresence>
                  
                  {editForm.socials.length === 0 && (
                    <div className="text-center py-8 bg-slate-50 border border-slate-100 border-dashed rounded-xl">
                      <Globe className="w-8 h-8 text-slate-300 mx-auto mb-2" />
                      <p className="text-sm font-medium text-slate-500">No links added yet.</p>
                      <p className="text-xs text-slate-400">Add your social media to let others connect with you.</p>
                    </div>
                  )}
                </div>
              </section>

            </div>
          </div>
          
          {/* Sticky Save / Cancel Footer */}
          <div className="sticky bottom-0 right-0 w-full bg-white/90 backdrop-blur-md border-t border-slate-100 p-4 sm:px-8 py-5 flex justify-end gap-3 z-50 rounded-b-[2rem] mt-auto">
            <Button variant="ghost" onClick={() => setIsDialogOpen(false)} className="rounded-full font-semibold">
              Cancel
            </Button>
            <Button onClick={handleSave} className="rounded-full font-semibold bg-indigo-600 hover:bg-indigo-700 text-white shadow-md shadow-indigo-200 px-8">
              Save Profile
            </Button>
          </div>

        </DialogContent>
      </Dialog>
    </div>
  );
}
